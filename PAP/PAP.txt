ALGORITHME "calcul aire cercle"
VARIABLES 
	rayon :REEL
	aire  :REEL
CONSTANTES
	Pi <- 3.14 :REEL
DEBUT
	AFFICHER ("entrez le rayon")
	LIRE rayon
	aire <- Pi*(rayon)^2
	AFFICHER("aire du cercle :"+aire)
FIN

ALGORITHME "divisible par 3"
VARIABLES
	nb :ENTIER
DEBUT
	AFFICHER ("entrez un nombre entier :")
	LIRE(nb)
	SI (nb mod 3) = 0 ALORS
		AFFICHER ("votre nombre est divisible par 3")
	SINON
		AFFICHER ("votre nombre n'est pas divisible par 3")
	FINSI
FIN

ALGORITHME "état par age"
VARIABLES
	age :ENTIER
DEBUT
	AFFICHER ("entrez l'age :")
	LIRE(age)
	AFFICHER (NEWLINE)
	
	SI (age >= 0) ET (age <= 3) ALORS
		AFFICHER ("la personne est un bébé")
	SINON
		SI (age >= 4 ) ET (age <= 11) ALORS	
			AFFICHER ("la personne est un enfant")
		SINON
			SI (age >=12) ET (age <= 17) ALORS
				AFFICHER ("la personne est un ado")
			SINON
				AFFICHER ("la personne est adulte")
			FINSI
		FINSI
	FINSI
FIN
